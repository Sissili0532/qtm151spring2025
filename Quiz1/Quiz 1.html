  • Print list_emissions

# Write your answer here:
datasets = ["data_quiz/china_co2_emissions.csv","data_quiz/japan_co2_emissions.csv"]

list_emissions = []

for data in datasets:
    df = pd.read_csv(data)

    mean_value = df["total_emissions"].mean()

    print(f"Mean of total_emissions in {data}: {mean_value:.2f}")

list_emissions.append(mean_value)
print(list_emissions)
